        setbuf(stdout, NULL);

        FILE* fptr = fopen("flag.txt", "r");

        if (fptr == NULL) {

        fscanf(fptr, "%s", flag);

def start(argv=[], *a, **kw):
    if args.GDB:  
        return gdb.debug([exe] + argv, gdbscript=gdbscript, *a, **kw)
    elif args.REMOTE: 
        return remote(sys.argv[1], sys.argv[2], *a, **kw)
    else: 
        return process([exe] + argv, *a, **kw)
